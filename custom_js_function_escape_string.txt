function jsonEscape(str) {
    return str
        .replace(/\\/g, '\\\\')
        .replace(/\"/g, '\\"')
        .replace(/\n/g, '\\n')
        .replace(/\r/g, '\\r')
        .replace(/\t/g, '\\t');
}

try {
    const text = await jsonEscape($args);
    return text;
} catch (error) {
    console.error(error);
    return '';
}
